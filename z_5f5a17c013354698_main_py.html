23def configure_logging(): 

24 """ 

25 Configure application logging. 

26 - LOG_LEVEL: default DEBUG 

27 - LOG_FILE: optional path to file (will use RotatingFileHandler) 

28 - If LOG_FILE is not set logs are streamed to stdout (recommended for Docker) 

29 """ 

30 log_level = os.getenv("LOG_LEVEL", "DEBUG").upper() 

31 log_file = os.getenv("LOG_FILE", "") 

32 

33 root_logger = logging.getLogger("bab") 

34 # Avoid duplicate handlers when this function is called multiple times 

35 if root_logger.handlers: 

36 for h in list(root_logger.handlers): 

37 root_logger.removeHandler(h) 

38 

39 root_logger.setLevel(getattr(logging, log_level, logging.DEBUG)) 

40 

41 formatter = logging.Formatter("%(asctime)s %(levelname)s [%(name)s] %(message)s") 

42 

43 if log_file: 

44 # Ensure directory exists 

45 try: 

46 Path(log_file).parent.mkdir(parents=True, exist_ok=True) 

47 except Exception: 

48 pass 

49 fh = RotatingFileHandler(log_file, maxBytes=10 * 1024 * 1024, backupCount=5) 

50 fh.setFormatter(formatter) 

51 root_logger.addHandler(fh) 

52 sh = logging.StreamHandler(sys.stdout) 

53 sh.setFormatter(formatter) 

54 root_logger.addHandler(sh) 

55 root_logger.debug( 

56 "Logging configured with level %s, output to %s", 

57 log_level, 

58 log_file or "stdout", 

59 )

62@app.on_event("startup") 

63async def startup_event(): 

64 """ 

65 On container start, restore per-user jobs from the configured data directory, 

66 and load persisted per-user tokens into the Firebase singleton so DB calls can use them. 

67 Jobs are scheduled daily at APP_JOB_HOUR:APP_JOB_MINUTE (defaults to 18:00). 

68 """ 

69 configure_logging() 

70 logger = logging.getLogger(__name__) 

71 logger.info("Starting application - restoring jobs and loading tokens") 

72 

73 base_data_dir = Path(os.getenv("APP_USER_DATA_DIR", "/var/app/user_data")) 

74 # Use explicit daily target hour/minute instead of an interval in seconds 

75 target_hour = int(os.getenv("APP_JOB_HOUR", "18")) 

76 target_minute = int(os.getenv("APP_JOB_MINUTE", "0")) 

77 

78 logger.debug("Using base data directory: %s", base_data_dir) 

79 logger.debug( 

80 "Scheduling jobs daily at %02d:%02d (local container time)", 

81 target_hour, 

82 target_minute, 

83 ) 

84 

85 # restore jobs (creates per-user folders if needed) using daily schedule 

86 scheduler.restore_jobs_from_dir(base_data_dir, target_hour, target_minute) 

87 logger.info("Restored scheduled jobs from %s", base_data_dir) 

88 

89 # populate firebase.user_tokens from disk and attempt token refresh 

90 try: 

91 firebase.load_tokens_from_dir(base_data_dir, refresh=True) 

92 logger.info("Loaded user tokens from %s", base_data_dir) 

93 except Exception as exc:
